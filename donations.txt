        // Format currency based on type
        function formatCurrency(amount, currency) {
            if (currency === "SYP") {
                return amount.toLocaleString('ar-SY') + " ل.س";
            } else if (currency === "USD") {
                return "$" + amount.toLocaleString('en-US');
            }
            return amount.toLocaleString('ar-SY');
        }

        // Format date to Arabic format
        function formatDate(dateString) {
            const date = new Date(dateString);
            return date.toLocaleDateString('ar-SA', {
                year: 'numeric',
                month: 'long',
                day: 'numeric'
            });
        }

        // Render donation cards
        function renderDonations(donations) {
            const container = document.getElementById('donationsContainer');
            
            if (donations.length === 0) {
                container.innerHTML = `
                    <div class="no-results">
                        <i class="fas fa-search"></i>
                        <h3>لا توجد تبرعات تطابق معايير البحث</h3>
                        <p>حاول استخدام فلتر مختلف أو ابحث مرة أخرى</p>
                    </div>
                `;
                return;
            }
            
            container.innerHTML = '';
            
            donations.forEach(donation => {
                const card = document.createElement('div');
                card.className = 'donation-card';
                card.innerHTML = `
                    <div class="card-header">
                        <h3 class="project-name">${donation.projectName}</h3>
                        <div class="donor-name">

        // Filter donations based on selected filter
        function filterDonations(filter) {
            let filteredDonations = [...donationsData];
            
            if (filter === 'completed') {
                filteredDonations = donationsData.filter(d => d.status === 'completed');
            } else if (filter === 'scheduled') {
                filteredDonations = donationsData.filter(d => d.status === 'scheduled');
            } else if (filter === 'large') {
                filteredDonations = donationsData.filter(d => d.amount > 1000000);
            }
            
            renderDonations(filteredDonations);
        }

                    // Remove active class from all buttons
                    document.querySelectorAll('.filter-btn').forEach(btn => {
                        btn.classList.remove('active');
                    });
                    
                    // Add active class to clicked button
                    this.classList.add('active');
                    
                    // Filter donations
                    const filter = this.getAttribute('data-filter');

                });
            });
